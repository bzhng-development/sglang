 49def get_token_num_per_seq(
 50    forward_mode: ForwardMode,
 51    spec_info: Optional[Union[EagleDraftInput, EagleVerifyInput]] = None,
 52):
 53    if forward_mode.is_target_verify():
 54        return spec_info.draft_token_num
 55    elif forward_mode.is_decode():
 56        return 1
 57    elif forward_mode.is_idle():
 58        return 0

 61        return None

 65def compute_split_seq_index(
 66    forward_mode: "ForwardMode",
 67    num_tokens: int,
 68    extend_lens: Optional[Sequence[int]],
 69    token_num_per_seq: Optional[int],
 70) -> Optional[int]:
 71    if forward_mode == ForwardMode.EXTEND:
 72        assert extend_lens is not None
 73        return _split_extend_seqs(extend_lens)
 74    elif forward_mode.is_target_verify() or forward_mode.is_decode():
 75        assert token_num_per_seq is not None
 76        return (num_tokens // token_num_per_seq) // 2
 77    elif forward_mode.is_idle():
 78        assert num_tokens == 0
 79        return 0
 80    else:
 81        raise NotImplementedError()

 84def _is_two_chunk_split_enabled(extend_lens: Sequence[int]) -> bool:
 85    if extend_lens is None:
 86        return False
 87
 88    vanilla_split_seq_index = _split_array_by_balanced_sum(extend_lens)
 89    left_sum = sum(extend_lens[:vanilla_split_seq_index])
 90    overall_sum = sum(extend_lens)
 91    threshold = get_tbo_token_distribution_threshold()
 92    assert threshold <= 0.5, f"{threshold=}"
 93    return left_sum < overall_sum * threshold or left_sum > overall_sum * (
 94        1 - threshold
 95    )

 98def _split_extend_seqs(arr: Sequence[int]) -> int:
 99    if _is_two_chunk_split_enabled(arr):
100        return _split_array_by_cum_less_than_half(arr)
101
102    return _split_array_by_balanced_sum(arr)

105def _split_array_by_cum_less_than_half(arr: Sequence[int]) -> int:
106    left_sum = 0
107    overall_sum = sum(arr)
108    half_sum = overall_sum // 2
109    chosen_index = 0
110
111    for i in range(len(arr)):
112        left_sum += arr[i]
113        if left_sum > half_sum:
114            chosen_index = i
115            break
116
117    return chosen_index

120def _split_array_by_balanced_sum(arr: Sequence[int]) -> int:
121    overall_sum = sum(arr)
122    left_sum = 0
123    min_diff = float("inf")
124    best_index = 0
125
126    for i in range(1, len(arr)):
127        left_sum += arr[i - 1]
128        right_sum = overall_sum - left_sum
129        diff = abs(left_sum - right_sum)
130        if diff <= min_diff:
131            min_diff = diff
132            best_index = i
133        else:
134            break
135
136    return best_index

139def _update_device_and_sum_field_from_cpu_field(
140    batch: ForwardBatch, cpu_field: str, device_field: str, sum_field: str = None
141):
142    cpu_value = getattr(batch, cpu_field, None)
143    old_device_value = getattr(batch, device_field, None)
144    if (
145        cpu_value is None
146        or old_device_value is None
147        or not (isinstance(cpu_value, torch.Tensor) or isinstance(cpu_value, list))
148    ):
149        return
150
151    new_device_value = (
152        cpu_value
153        if isinstance(cpu_value, torch.Tensor)
154        else torch.tensor(cpu_value, dtype=old_device_value.dtype)
155    ).to(device=global_server_args_dict["device"], non_blocking=True)
156    setattr(batch, device_field, new_device_value)
157
158    if sum_field is not None:
159        sum_value = (
160            cpu_value.sum().item()
161            if isinstance(cpu_value, torch.Tensor)
162            else sum(cpu_value)
163        )
164        setattr(batch, sum_field, sum_value)

167def _compute_mask_offset(seq_index: int, spec_info: Optional[EagleVerifyInput]) -> int:
168    if seq_index == 0:
169        return 0
170
171    offset = 0
172    max_seq_len = min(seq_index, spec_info.seq_lens_cpu.shape[0])
173    for i in range(max_seq_len):
174        offset += (
175            spec_info.seq_lens_cpu[i] + spec_info.draft_token_num
176        ) * spec_info.draft_token_num
177    return offset

180def split_spec_info(
181    spec_info: Optional[EagleVerifyInput],
182    start_seq_index: int,
183    end_seq_index: int,
184    start_token_index: int,
185    end_token_index: int,
186):
187    if spec_info is None:
188        return None
189    if spec_info.draft_token is not None:
190        draft_token = spec_info.draft_token[start_token_index:end_token_index]
191    else:
192        draft_token = None
193    if spec_info.custom_mask is not None and spec_info.draft_token is not None:
194        custom_mask_start = _compute_mask_offset(start_seq_index, spec_info)
195        if end_seq_index == spec_info.seq_lens_cpu.shape[0]:
196            custom_mask_end = spec_info.custom_mask.shape[0]
197        else:
198            custom_mask_end = _compute_mask_offset(end_seq_index, spec_info)
199
200        if custom_mask_end > custom_mask_start:
201            custom_mask = spec_info.custom_mask[custom_mask_start:custom_mask_end]
202        else:
203            custom_mask = spec_info.custom_mask
204    else:
205        custom_mask = spec_info.custom_mask
206    if spec_info.positions is not None:
207        positions = spec_info.positions[start_token_index:end_token_index]
208    else:
209        positions = None
210    if spec_info.retrive_index is not None:
211        retrive_index = spec_info.retrive_index[start_seq_index:end_seq_index]
212    else:
213        retrive_index = None
214    if spec_info.retrive_next_token is not None:
215        retrive_next_token = spec_info.retrive_next_token[start_seq_index:end_seq_index]
216    else:
217        retrive_next_token = None
218    if spec_info.retrive_next_sibling is not None:
219        retrive_next_sibling = spec_info.retrive_next_sibling[
220            start_seq_index:end_seq_index
221        ]
222    else:
223        retrive_next_sibling = None
224    if spec_info.retrive_cum_len is not None:
225        retrive_cum_len = spec_info.retrive_cum_len[start_seq_index:end_seq_index]
226    else:
227        retrive_cum_len = None
228
229    if spec_info.seq_lens_cpu is not None:
230        seq_lens_cpu = spec_info.seq_lens_cpu[start_seq_index:end_seq_index]
231    else:
232        seq_lens_cpu = None
233    if seq_lens_cpu is not None:
234        seq_lens_sum = seq_lens_cpu.sum()
235    else:
236        seq_lens_sum = None
237    output_spec_info = replace(
238        spec_info,
239        custom_mask=custom_mask,
240        draft_token=draft_token,
241        positions=positions,
242        retrive_index=retrive_index,
243        retrive_next_token=retrive_next_token,
244        retrive_next_sibling=retrive_next_sibling,
245        retrive_cum_len=retrive_cum_len,
246        seq_lens_cpu=seq_lens_cpu,
247        seq_lens_sum=seq_lens_sum,
248    )
249    return output_spec_info

252def compute_split_token_index(
253    split_seq_index: int,
254    forward_mode: "ForwardMode",
255    extend_seq_lens: Optional[Sequence[int]],
256    token_num_per_seq: Optional[int],
257) -> int:
258    if forward_mode == ForwardMode.EXTEND:
259        assert extend_seq_lens is not None
260        if _is_two_chunk_split_enabled(extend_seq_lens):
261            return sum(extend_seq_lens) // 2
262        return sum(extend_seq_lens[:split_seq_index])
263    elif forward_mode.is_target_verify() or forward_mode.is_decode():
264        assert token_num_per_seq is not None
265        return split_seq_index * token_num_per_seq
266    elif forward_mode.is_idle():
267        assert split_seq_index == 0
268        return 0
269    else:
270        raise NotImplementedError

277):

291        split_seq_index=tbo_split_seq_index,
292        forward_mode=forward_mode_for_tbo_split,
293        extend_seq_lens=None,
294        token_num_per_seq=token_num_per_seq,
295    )
296    return tbo_split_seq_index, tbo_split_token_index

606        num_tokens = batch.input_ids.shape[0]

608

610
611        for key in [
612            "input_ids",
613            "positions",
614            "out_cache_loc",
615        ]:
616            old_value = getattr(batch, key)
617            assert (
618                old_value.shape[0] == num_tokens
619            ), f"{key=} {old_value=} {num_tokens=} {batch=}"

631            "extend_logprob_start_lens_cpu",
632            "lora_ids",
633        ]:
634            old_value = getattr(batch, key)
635            if old_value is None:
636                continue

638                key == "extend_seq_lens"
639                or key == "extend_prefix_lens"
640                or key == "extend_start_loc"
641                or key == "extend_prefix_lens_cpu"
642                or key == "extend_seq_lens_cpu"
643                or key == "extend_logprob_start_lens_cpu"

648                len(old_value) == num_seqs
649            ), f"{key=} {old_value=} {num_seqs=} {batch=}"
650            output_dict[key] = old_value[start_seq_index:end_seq_index]
651
652        spec_info = getattr(batch, "spec_info")
653        output_spec_info = split_spec_info(
654            spec_info=spec_info,
655            start_token_index=start_token_index,

661        for key in [
662            "forward_mode",
663            "is_extend_in_batch",
664            "return_logprob",
665            "req_to_token_pool",
666            "token_to_kv_pool",
667            "can_run_dp_cuda_graph",
668            "dp_padding_mode",
669            "global_forward_mode",
670            "spec_algorithm",

674            "split_index",  # for split prefill
675            "orig_seq_lens",  # only used by qwen-1m, thus not care
676        ]:
677            output_dict[key] = getattr(batch, key)
678        if not batch.forward_mode.is_target_verify():

680                _compute_extend_num_tokens(batch.input_ids, batch.forward_mode)

688        if (
689            global_server_args_dict["moe_dense_tp_size"] == 1
690            and batch.global_dp_buffer_len is not None
691        ):
692            sum_len = end_token_index - start_token_index
693            global_dp_buffer_len = sum_len
694        else:
695            global_dp_buffer_len = None
696
697        output_dict.update(
698            dict(
699                batch_size=end_seq_index - start_seq_index,
700                seq_lens_sum=(
701                    output_dict["seq_lens_cpu"].sum()
702                    if "seq_lens_cpu" in output_dict
703                    else None
704                ),
705                extend_num_tokens=extend_num_tokens,
706                attn_backend=output_attn_backend,
707                num_token_non_padded=out_num_token_non_padded,
708                tbo_split_seq_index=None,
709                tbo_parent_token_range=(start_token_index, end_token_index),
710                tbo_children=None,
711                global_num_tokens_gpu=None,
712                global_num_tokens_cpu=None,
713                global_dp_buffer_len=global_dp_buffer_len,
714                global_num_tokens_for_logprob_gpu=None,
715                global_num_tokens_for_logprob_cpu=None,
716                sampling_info=None,
717                # For logits and logprobs post processing, thus we do not care
718                temp_scaled_logprobs=False,
719                temperature=None,
720                top_p_normalized_logprobs=False,
721                top_p=None,
722                mm_inputs=None,
723                top_logprobs_nums=None,
724                token_ids_logprobs=None,
725                next_token_logits_buffer=None,

729        errors = []

734                )

736            raise Exception(f"{len(errors)} errors happen:\n" + "\n\n".join(errors))
