127    def _handle_dtype_to_regex(self, sampling_params: SglSamplingParams):
128        if sampling_params.dtype is None:
129            return
130
131        if sampling_params.stop == ():
132            sampling_params.stop = []
133
134        dtype_regex = None
135        if sampling_params.dtype in ["int", int]:
136
137            dtype_regex = REGEX_INT
138            sampling_params.stop.extend([" ", "\n"])
139        elif sampling_params.dtype in ["float", float]:
140
141            dtype_regex = REGEX_FLOAT
142            sampling_params.stop.extend([" ", "\n"])
143        elif sampling_params.dtype in ["str", str]:
144
145            dtype_regex = REGEX_STR
146        elif sampling_params.dtype in ["bool", bool]:
147
148            dtype_regex = REGEX_BOOL
149        else:
150            raise RuntimeError(f"Invalid dtype: {sampling_params.dtype}")
151
152        if dtype_regex is not None and sampling_params.regex is not None:
153            warnings.warn(
154                f"Both dtype and regex are set. Only dtype will be used. dtype: {sampling_params.dtype}, regex: {sampling_params.regex}"
155            )
156
157        sampling_params.regex = dtype_regex

248    def select(
249        self,
250        s: StreamExecutor,
251        choices: List[str],
252        temperature: float,
253        choices_method: ChoicesSamplingMethod,
254    ) -> ChoicesDecision:
255        assert temperature <= 1e-5
256
257        # Cache common prefix
258        data = {"text": s.text_, "sampling_params": {"max_new_tokens": 0}}
259        obj = self._generate_http_request(s, data)
260        prompt_len = obj["meta_info"]["prompt_tokens"]
261        logprob_start_len = max(prompt_len - 2, 0)  # For token healing
262
263        # Compute logprob
264        data = {
265            "text": [s.text_ + c for c in choices],
266            "sampling_params": {
267                "max_new_tokens": 0,
268                "temperature": 0,
269            },
270            "return_logprob": True,
271            "return_text_in_logprobs": True,
272            "logprob_start_len": logprob_start_len,
273        }
274        obj = self._generate_http_request(s, data)
275
276        input_token_logprobs = [r["meta_info"]["input_token_logprobs"] for r in obj]
277        output_token_logprobs = [r["meta_info"]["output_token_logprobs"] for r in obj]
278        normalized_prompt_logprobs = [
279            compute_normalized_prompt_logprobs(r["meta_info"]["input_token_logprobs"])
280            for r in obj
281        ]
282
283        # Remove extra token if no token healing occurred
284        for i in range(len(input_token_logprobs)):
285            healed_token_str = input_token_logprobs[i][0][-1]
286            if s.text_.endswith(healed_token_str):
287                healed_token_logprob = input_token_logprobs[i][0][0]
288                normalized_prompt_logprobs[i] = (
289                    normalized_prompt_logprobs[i] * len(input_token_logprobs[i])
290                    - healed_token_logprob
291                ) / (len(input_token_logprobs[i]) - 1)
292                input_token_logprobs[i] = input_token_logprobs[i][1:]
293
294        # Compute unconditional logprobs if required
295        if choices_method.requires_unconditional_logprobs:
296            input_ids = [[el[1] for el in subl] for subl in input_token_logprobs]
297            data = {
298                "input_ids": input_ids,
299                "sampling_params": {"max_new_tokens": 0},
300                "return_logprob": True,
301            }
302            obj = self._generate_http_request(s, data)
303            unconditional_token_logprobs = [
304                r["meta_info"]["input_token_logprobs"] for r in obj
305            ]
306        else:
307            unconditional_token_logprobs = None
308
309        return choices_method(
310            choices=choices,
311            normalized_prompt_logprobs=normalized_prompt_logprobs,
312            input_token_logprobs=input_token_logprobs,
313            output_token_logprobs=output_token_logprobs,
314            unconditional_token_logprobs=unconditional_token_logprobs,
315        )

248    def select(
249        self,
250        s: StreamExecutor,
251        choices: List[str],
252        temperature: float,
253        choices_method: ChoicesSamplingMethod,
254    ) -> ChoicesDecision:
255        assert temperature <= 1e-5
256
257        # Cache common prefix
258        data = {"text": s.text_, "sampling_params": {"max_new_tokens": 0}}
259        obj = self._generate_http_request(s, data)
260        prompt_len = obj["meta_info"]["prompt_tokens"]
261        logprob_start_len = max(prompt_len - 2, 0)  # For token healing
262
263        # Compute logprob
264        data = {
265            "text": [s.text_ + c for c in choices],
266            "sampling_params": {
267                "max_new_tokens": 0,
268                "temperature": 0,
269            },
270            "return_logprob": True,
271            "return_text_in_logprobs": True,
272            "logprob_start_len": logprob_start_len,
273        }
274        obj = self._generate_http_request(s, data)
275
276        input_token_logprobs = [r["meta_info"]["input_token_logprobs"] for r in obj]
277        output_token_logprobs = [r["meta_info"]["output_token_logprobs"] for r in obj]
278        normalized_prompt_logprobs = [
279            compute_normalized_prompt_logprobs(r["meta_info"]["input_token_logprobs"])
280            for r in obj
281        ]
282
283        # Remove extra token if no token healing occurred
284        for i in range(len(input_token_logprobs)):
285            healed_token_str = input_token_logprobs[i][0][-1]
286            if s.text_.endswith(healed_token_str):
287                healed_token_logprob = input_token_logprobs[i][0][0]
288                normalized_prompt_logprobs[i] = (
289                    normalized_prompt_logprobs[i] * len(input_token_logprobs[i])
290                    - healed_token_logprob
291                ) / (len(input_token_logprobs[i]) - 1)
292                input_token_logprobs[i] = input_token_logprobs[i][1:]
293
294        # Compute unconditional logprobs if required
295        if choices_method.requires_unconditional_logprobs:
296            input_ids = [[el[1] for el in subl] for subl in input_token_logprobs]
297            data = {
298                "input_ids": input_ids,
299                "sampling_params": {"max_new_tokens": 0},
300                "return_logprob": True,
301            }
302            obj = self._generate_http_request(s, data)
303            unconditional_token_logprobs = [
304                r["meta_info"]["input_token_logprobs"] for r in obj
305            ]
306        else:
307            unconditional_token_logprobs = None
308
309        return choices_method(
310            choices=choices,
311            normalized_prompt_logprobs=normalized_prompt_logprobs,
312            input_token_logprobs=input_token_logprobs,
313            output_token_logprobs=output_token_logprobs,
314            unconditional_token_logprobs=unconditional_token_logprobs,
315        )
